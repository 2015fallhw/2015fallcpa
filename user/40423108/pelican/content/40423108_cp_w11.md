猜數字程式碼:
~~~html

#coding: utf-8
# 猜數字遊戲
import random
   
x = int(input("請輸入下限的整數:"))
y = int(input("請輸入上限的整數:"))

標準答案 = random.randint(x,y)
print("標準答案:",標準答案)
PC猜的數字 = random.randint(x,y)
print("PC猜的數字:",PC猜的數字)
猜測次數 = 1
while 標準答案 != PC猜的數字:
    if 標準答案 < PC猜的數字:
        print("太大了，再猜一次 :)加油")
        y = PC猜的數字-1
    else:
        print("太小了，再猜一次 :)加油")
        x = PC猜的數字+1
    PC猜的數字= random.randint(x,y)
    print("PC猜的數字:",PC猜的數字)
    猜測次數 += 1
   
print("猜對了！總共猜了", 猜測次數, "次")
~~~
