<canvas id="plotarea2" width="500" height="500"></canvas>

<script type="text/python3">
# 導入 doc
from browser import document as doc
from browser import console
import math

# 準備繪圖畫布
canvas = doc["plotarea2"]
ctx = canvas.getContext("2d")

# 開始畫直線
ctx.beginPath()
ctx.lineWidth = 5
ctx.moveTo(250,100)
ctx.lineTo(150,200)
ctx.lineTo(200,200)
ctx.lineTo(100,300)
ctx.lineTo(200,300)
ctx.lineTo(200,400)
ctx.lineTo(300,400)
ctx.lineTo(300,300)
ctx.lineTo(400,300)
ctx.lineTo(300,200)
ctx.lineTo(350,200)
ctx.lineTo(250,100)
ctx.strokeStyle = "#006000"
ctx.stroke()
</script>
